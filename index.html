تنبيه: تعذر جلب المشاريع السحابية من قاعدة البيانات

تم عرض المشاريع الافتراضية محلياً. قد تكون الخدمة السحابية (Supabase) غير متوفرة حالياً أو متوقفة مؤقتاً.

💡 حل المشكلة: يرجى نسخ الكود التالي وتشغيله في الـ SQL Editor الخاص بـ Supabase لإنشاء الجداول والسياسات: 

-- 1. Create projects table
create table if not exists public.projects (
  id uuid default gen_random_uuid() primary key,
  title text not null,
  description text not null,
  live_url text not null,
  github_url text not null,
  image_url text,
  lang text not null,
  email text not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 2. Create profiles table
create table if not exists public.profiles (
  id uuid default gen_random_uuid() primary key,
  email text unique not null,
  username text unique not null,
  display_name text,
  avatar_url text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 3. Insert admin profile (modify with your email if needed)
insert into public.profiles (email, username, display_name)
values ('noviumnodes@gmail.com', 'noviumnodes', 'NoviumNodes Team')
on conflict (email) do nothing;

-- 4. Enable Row Level Security (RLS)
alter table public.projects enable row level security;
alter table public.profiles enable row level security;

-- 5. Create Policies (Allow anyone to read, only authenticated users to modify)
create policy "Allow public read projects" on public.projects for select using (true);
create policy "Allow public read profiles" on public.profiles for select using (true);

create policy "Allow admin write projects" on public.projects 
  for all using (auth.role() = 'authenticated');

create policy "Allow admin write profiles" on public.profiles 
  for all using (auth.role() = 'authenticated');

⚠️ تأكد أيضاً من إنشاء Storage Bucket باسم project-images وجعله Public لرفع صور غلاف المشاريع والصور الشخصية.
